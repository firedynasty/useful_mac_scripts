on run {input, parameters}
    if (count of input) is 0 then
        display dialog "No files were selected." buttons {"OK"} default button "OK"
        return
    end if

    set shellScript to ""
    set pyScript to "/Users/stanleytan/Documents/technical/python/convert_html_to_docx/02evernote/csvToHtml.py"

    repeat with selectedFile in input
        set filePath to POSIX path of selectedFile

        -- Swap .csv extension for .html
        set outputPath to do shell script "echo " & quoted form of filePath & " | sed 's/\\.csv$/.html/'"

        set shellScript to shellScript & "echo '=== Converting: " & filePath & " ===' && "
        set shellScript to shellScript & "/Users/stanleytan/anaconda3/bin/python3 " & quoted form of pyScript & " " & quoted form of filePath & " " & quoted form of outputPath & " ; "
    end repeat

    tell application "Terminal"
        activate
        do script shellScript in front window
    end tell
end run
