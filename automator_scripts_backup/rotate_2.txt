on run {input, parameters}
    if (count of input) is 0 then
        display dialog "No files were selected." buttons {"OK"} default button "OK"
        return
    end if

    set userChoice to text returned of (display dialog "Leave empty to generate test clips (60 sec at 90/180/270)." & return & return & "Or enter 90, 180, or 270 to rotate full video." default answer "" buttons {"Cancel", "OK"} default button "OK")

    repeat with selectedFile in input
        set filePath to POSIX path of selectedFile
        set baseName to do shell script "f=" & quoted form of filePath & "; echo \"${f%.*}\""
        set ext to do shell script "f=" & quoted form of filePath & "; echo \"${f##*.}\""

        if userChoice is "" then
            -- Generate 3 test clips (60 seconds each)
            set shellScript to "/opt/homebrew/bin/ffmpeg -nostdin -y -i " & quoted form of filePath & " -t 60 -vf transpose=1 -c:v hevc_videotoolbox -c:a copy " & quoted form of (baseName & "_test90." & ext) & " && " & ¬
                "/opt/homebrew/bin/ffmpeg -nostdin -y -i " & quoted form of filePath & " -t 60 -vf 'transpose=1,transpose=1' -c:v hevc_videotoolbox -c:a copy " & quoted form of (baseName & "_test180." & ext) & " && " & ¬
                "/opt/homebrew/bin/ffmpeg -nostdin -y -i " & quoted form of filePath & " -t 60 -vf transpose=2 -c:v hevc_videotoolbox -c:a copy " & quoted form of (baseName & "_test270." & ext) & " && " & ¬
                "open " & quoted form of (baseName & "_test90." & ext) & " && " & ¬
                "open " & quoted form of (baseName & "_test180." & ext) & " && " & ¬
                "open " & quoted form of (baseName & "_test270." & ext) & " && " & ¬
                "echo '=== Test clips ready — preview them, then run again with 90, 180, or 270 ==='"

            tell application "Terminal"
                activate
                do script shellScript in front window
            end tell

        else if userChoice is "90" or userChoice is "180" or userChoice is "270" then
            -- Apply chosen rotation to full video
            if userChoice is "90" then
                set vf to "transpose=1"
            else if userChoice is "180" then
                set vf to "transpose=1,transpose=1"
            else
                set vf to "transpose=2"
            end if

            set outFile to do shell script "f=" & quoted form of (baseName & "_rotated." & ext) & "; n=1; out=\"$f\"; while [ -e \"$out\" ]; do out='" & baseName & "_rotated_'\"$n\".'" & "." & ext & "'; n=$((n+1)); done; echo \"$out\""

            set shellScript to "/opt/homebrew/bin/ffmpeg -nostdin -i " & quoted form of filePath & " -vf " & quoted form of vf & " -c:v hevc_videotoolbox -c:a copy " & quoted form of outFile & " && " & ¬
                "rm -f " & quoted form of (baseName & "_test90." & ext) & " " & quoted form of (baseName & "_test180." & ext) & " " & quoted form of (baseName & "_test270." & ext) & " && " & ¬
                "echo '=== Done — test clips cleaned up ==='"

            tell application "Terminal"
                activate
                do script shellScript in front window
            end tell

        else
            display dialog "Invalid choice. Enter 90, 180, or 270." buttons {"OK"} default button "OK"
        end if
    end repeat
end run
