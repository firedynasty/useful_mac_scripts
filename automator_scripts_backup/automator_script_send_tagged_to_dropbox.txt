on run {input, parameters}
    if (count of input) is 0 then
        display dialog "No files were selected." buttons {"OK"} default button "OK"
        return
    end if

    set dropboxDest to text returned of (display dialog "Dropbox destination path:" default answer "/notes")

    set folderPath to POSIX path of (item 1 of input)
    set shellScript to "/Users/stanleytan/anaconda3/bin/python3 /Users/stanleytan/Documents/technical/python/file_sorter/tagged_sync.py " & quoted form of folderPath & " \"dropbox:" & dropboxDest & "\" --no-index"

    tell application "Terminal"
        activate
        do script shellScript in front window
    end tell
end run
